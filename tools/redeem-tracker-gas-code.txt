/**
 * 頭竹苗福利社｜福利卡使用紀錄資料庫 + 數據後台
 *
 * 用途：
 * 1. 建立 Google Sheet 分頁與欄位
 * 2. 接收官網 redeem.html 送來的「店家確認使用」紀錄
 * 3. 用紀錄編號 event_id 去重，避免重複寫入
 * 4. 自動整理 dashboard、店家、日期、地區、分類、來源統計
 *
 * 第一次使用請先改：
 * const SPREADSHEET_ID = '請填入你的 Google Sheet ID';
 *
 * 建議執行順序：
 * 1. setupRedeemTrackingDatabase()
 * 2. 部署 Web App
 * 3. 將 Web App URL 填回官網 redeem.html 的 window.TZM_REDEEM_LOG_URL
 */

const SPREADSHEET_ID = '請填入你的 Google Sheet ID';

const SHEETS = {
  logs: 'redeem_logs',
  dashboard: 'dashboard',
  shopSummary: 'shop_summary',
  dailySummary: 'daily_summary',
  categorySummary: 'category_summary',
  areaSummary: 'area_summary',
  sourceSummary: 'source_summary'
};

const LOG_HEADERS = [
  '寫入時間',
  '紀錄編號',
  '事件類型',
  '確認時間',
  '店家ID',
  '店家名稱',
  '分類',
  '地區',
  '優惠內容',
  '來源',
  '入口位置',
  '匿名用戶ID',
  '頁面網址',
  'User Agent'
];

function onOpen() {
  SpreadsheetApp.getUi()
    .createMenu('頭竹苗福利社')
    .addItem('初始化資料庫分頁', 'setupRedeemTrackingDatabase')
    .addItem('重建統計後台', 'rebuildRedeemDashboard')
    .addToUi();
}

function setupRedeemTrackingDatabase() {
  const ss = getSpreadsheet_();

  const logs = getOrCreateSheet_(ss, SHEETS.logs);
  forceHeader_(logs, LOG_HEADERS);
  logs.setFrozenRows(1);
  formatHeader_(logs, LOG_HEADERS.length, '#4B2E4F');
  logs.autoResizeColumns(1, LOG_HEADERS.length);

  setupSummarySheet_(ss, SHEETS.dashboard, ['區塊', '項目', '數值', '說明']);
  setupSummarySheet_(ss, SHEETS.shopSummary, ['排名', '店家ID', '店家名稱', '分類', '地區', '確認使用次數', '最後使用時間']);
  setupSummarySheet_(ss, SHEETS.dailySummary, ['日期', '確認使用次數', '不重複匿名用戶數']);
  setupSummarySheet_(ss, SHEETS.categorySummary, ['分類', '確認使用次數', '不重複店家數']);
  setupSummarySheet_(ss, SHEETS.areaSummary, ['地區', '確認使用次數', '不重複店家數']);
  setupSummarySheet_(ss, SHEETS.sourceSummary, ['來源', '確認使用次數', '不重複匿名用戶數']);

  rebuildRedeemDashboard();
}

function doPost(e) {
  try {
    const data = parsePayload_(e);
    const result = appendRedeemLog_(data);
    rebuildRedeemDashboard();
    return json_({
      ok: true,
      message: result.duplicate ? 'duplicate ignored' : 'redeem log saved',
      duplicate: result.duplicate,
      event_id: result.event_id
    });
  } catch (error) {
    return json_({ ok: false, error: String(error) });
  }
}

function doGet(e) {
  const action = e && e.parameter ? e.parameter.action : '';

  if (action === 'setup') {
    setupRedeemTrackingDatabase();
    return json_({ ok: true, message: 'database setup completed' });
  }

  if (action === 'rebuild') {
    rebuildRedeemDashboard();
    return json_({ ok: true, message: 'dashboard rebuilt' });
  }

  return json_({
    ok: true,
    service: 'tzm_redeem_tracker',
    message: '頭竹苗福利社福利卡使用紀錄 API 正常'
  });
}

function appendRedeemLog_(data) {
  const ss = getSpreadsheet_();
  const sheet = getOrCreateSheet_(ss, SHEETS.logs);
  forceHeader_(sheet, LOG_HEADERS);

  const eventId = data.event_id || makeFallbackEventId_(data);
  if (eventIdExists_(sheet, eventId)) {
    return { duplicate: true, event_id: eventId };
  }

  sheet.appendRow([
    new Date(),
    eventId,
    data.event_type || 'redeem_confirmed',
    data.confirmed_at || '',
    data.shop_id || '',
    data.shop_name || '',
    data.category || '',
    data.area || '',
    data.offer || '',
    data.source || '',
    data.from || '',
    data.client_id || '',
    data.page_url || '',
    data.user_agent || ''
  ]);

  return { duplicate: false, event_id: eventId };
}

function rebuildRedeemDashboard() {
  const ss = getSpreadsheet_();
  const logsSheet = getOrCreateSheet_(ss, SHEETS.logs);
  forceHeader_(logsSheet, LOG_HEADERS);

  const values = logsSheet.getDataRange().getValues();
  const rows = values.slice(1).filter(row => row[1]);
  const confirmedRows = rows.filter(row => String(row[2] || '') === 'redeem_confirmed');

  const now = new Date();
  const todayText = formatDate_(now);
  const monthText = Utilities.formatDate(now, Session.getScriptTimeZone(), 'yyyy-MM');

  const total = confirmedRows.length;
  const todayCount = confirmedRows.filter(row => formatDate_(row[0]) === todayText).length;
  const monthCount = confirmedRows.filter(row => formatMonth_(row[0]) === monthText).length;
  const uniqueUsers = new Set(confirmedRows.map(row => row[11]).filter(Boolean)).size;
  const uniqueShops = new Set(confirmedRows.map(row => row[4]).filter(Boolean)).size;
  const latest = confirmedRows.length ? confirmedRows[confirmedRows.length - 1][0] : '';

  const topShop = firstOrEmpty_(buildShopSummary_(confirmedRows));
  const topCategory = firstOrEmpty_(buildSimpleSummary_(confirmedRows, 6, 4));
  const topArea = firstOrEmpty_(buildSimpleSummary_(confirmedRows, 7, 4));
  const topSource = firstOrEmpty_(buildSimpleSummary_(confirmedRows, 9, 11));

  writeRows_(ss, SHEETS.dashboard, ['區塊', '項目', '數值', '說明'], [
    ['總覽', '總確認使用次數', total, '店家按下「店家確認使用」的總次數'],
    ['總覽', '今日使用次數', todayCount, `今天 ${todayText} 的確認使用次數`],
    ['總覽', '本月使用次數', monthCount, `本月 ${monthText} 的確認使用次數`],
    ['總覽', '不重複匿名用戶數', uniqueUsers, '用瀏覽器匿名 ID 粗估，不等於 LINE 會員數'],
    ['總覽', '有使用紀錄的店家數', uniqueShops, '至少被確認使用一次的店家數'],
    ['總覽', '最後一筆寫入時間', latest, '最近一次寫入 Google Sheet 的時間'],
    ['排行', '目前最常使用店家', topShop[2] || '', topShop[5] ? `${topShop[5]} 次` : '尚無資料'],
    ['排行', '目前熱門分類', topCategory[0] || '', topCategory[1] ? `${topCategory[1]} 次` : '尚無資料'],
    ['排行', '目前熱門地區', topArea[0] || '', topArea[1] ? `${topArea[1]} 次` : '尚無資料'],
    ['排行', '目前主要來源', topSource[0] || '', topSource[1] ? `${topSource[1]} 次` : '尚無資料'],
    ['系統', '資料更新時間', new Date(), '最近一次重建後台的時間']
  ]);

  writeRows_(ss, SHEETS.shopSummary, ['排名', '店家ID', '店家名稱', '分類', '地區', '確認使用次數', '最後使用時間'], buildShopSummary_(confirmedRows));
  writeRows_(ss, SHEETS.dailySummary, ['日期', '確認使用次數', '不重複匿名用戶數'], buildDailySummary_(confirmedRows));
  writeRows_(ss, SHEETS.categorySummary, ['分類', '確認使用次數', '不重複店家數'], buildSimpleSummary_(confirmedRows, 6, 4));
  writeRows_(ss, SHEETS.areaSummary, ['地區', '確認使用次數', '不重複店家數'], buildSimpleSummary_(confirmedRows, 7, 4));
  writeRows_(ss, SHEETS.sourceSummary, ['來源', '確認使用次數', '不重複匿名用戶數'], buildSimpleSummary_(confirmedRows, 9, 11));
}

function buildShopSummary_(rows) {
  const map = {};
  rows.forEach(row => {
    const shopId = row[4] || 'unknown';
    if (!map[shopId]) {
      map[shopId] = {
        shopId,
        name: row[5] || '',
        category: row[6] || '',
        area: row[7] || '',
        count: 0,
        latest: ''
      };
    }
    map[shopId].count += 1;
    map[shopId].latest = row[0] || map[shopId].latest;
  });

  return Object.values(map)
    .sort((a, b) => b.count - a.count)
    .map((item, index) => [
      index + 1,
      item.shopId,
      item.name,
      item.category,
      item.area,
      item.count,
      item.latest
    ]);
}

function buildDailySummary_(rows) {
  const map = {};
  rows.forEach(row => {
    const date = formatDate_(row[0]);
    if (!map[date]) {
      map[date] = { date, count: 0, users: new Set() };
    }
    map[date].count += 1;
    if (row[11]) map[date].users.add(row[11]);
  });

  return Object.values(map)
    .sort((a, b) => String(b.date).localeCompare(String(a.date)))
    .map(item => [item.date, item.count, item.users.size]);
}

function buildSimpleSummary_(rows, keyIndex, uniqueIndex) {
  const map = {};
  rows.forEach(row => {
    const key = row[keyIndex] || '未分類';
    if (!map[key]) {
      map[key] = { key, count: 0, unique: new Set() };
    }
    map[key].count += 1;
    if (row[uniqueIndex]) map[key].unique.add(row[uniqueIndex]);
  });

  return Object.values(map)
    .sort((a, b) => b.count - a.count)
    .map(item => [item.key, item.count, item.unique.size]);
}

function eventIdExists_(sheet, eventId) {
  if (!eventId || sheet.getLastRow() < 2) return false;
  const ids = sheet.getRange(2, 2, sheet.getLastRow() - 1, 1).getValues().flat();
  return ids.some(id => String(id) === String(eventId));
}

function makeFallbackEventId_(data) {
  return [
    'redeem',
    data.shop_id || 'unknown',
    data.client_id || 'anon',
    data.confirmed_at || new Date().toISOString()
  ].join('_');
}

function parsePayload_(e) {
  if (!e || !e.postData || !e.postData.contents) {
    throw new Error('缺少 POST body');
  }
  return JSON.parse(e.postData.contents);
}

function getSpreadsheet_() {
  if (!SPREADSHEET_ID || SPREADSHEET_ID === '請填入你的 Google Sheet ID') {
    throw new Error('請先填入 SPREADSHEET_ID');
  }
  return SpreadsheetApp.openById(SPREADSHEET_ID);
}

function getOrCreateSheet_(ss, name) {
  return ss.getSheetByName(name) || ss.insertSheet(name);
}

function setupSummarySheet_(ss, name, headers) {
  const sheet = getOrCreateSheet_(ss, name);
  forceHeader_(sheet, headers);
  sheet.setFrozenRows(1);
  formatHeader_(sheet, headers.length, '#7D8F72');
  sheet.autoResizeColumns(1, headers.length);
}

function forceHeader_(sheet, headers) {
  sheet.getRange(1, 1, 1, headers.length).setValues([headers]);
  const extraColumns = sheet.getLastColumn() - headers.length;
  if (extraColumns > 0) {
    sheet.getRange(1, headers.length + 1, 1, extraColumns).clearContent();
  }
}

function writeRows_(ss, sheetName, headers, rows) {
  const sheet = getOrCreateSheet_(ss, sheetName);
  sheet.clearContents();
  sheet.getRange(1, 1, 1, headers.length).setValues([headers]);
  if (rows.length) {
    sheet.getRange(2, 1, rows.length, headers.length).setValues(rows);
  }
  sheet.setFrozenRows(1);
  formatHeader_(sheet, headers.length, '#4B2E4F');
  sheet.autoResizeColumns(1, headers.length);
}

function formatHeader_(sheet, length, color) {
  sheet.getRange(1, 1, 1, length)
    .setFontWeight('bold')
    .setBackground(color)
    .setFontColor('#FFFFFF');
}

function formatDate_(value) {
  const date = normalizeDate_(value);
  if (!date) return '未知日期';
  return Utilities.formatDate(date, Session.getScriptTimeZone(), 'yyyy-MM-dd');
}

function formatMonth_(value) {
  const date = normalizeDate_(value);
  if (!date) return '未知月份';
  return Utilities.formatDate(date, Session.getScriptTimeZone(), 'yyyy-MM');
}

function normalizeDate_(value) {
  if (!value) return null;
  const date = value instanceof Date ? value : new Date(value);
  return isNaN(date.getTime()) ? null : date;
}

function firstOrEmpty_(rows) {
  return rows && rows.length ? rows[0] : [];
}

function json_(payload) {
  return ContentService
    .createTextOutput(JSON.stringify(payload))
    .setMimeType(ContentService.MimeType.JSON);
}
